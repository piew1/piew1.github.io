} // Match face descriptor let bestMatch = null, bestDist = 0.6; for (let i = 1; i < rows.length; i++) { const r = rows[i]; if (r[10] !== 'approved' || !r[9]) continue; const stored = JSON.parse(r[9] || '[]'); logActivity(bestMatch[3], 'LOGIN', `สแกนใบหน้า (dist:${bestDist.toFixed(3)})`, data.device, data.ip); return { success: true, distance: dist, user: { id: bestMatch[0], firstName: bestMatch[1], lastName: bestMatch[2], username: bestMatch[3], email: bestMatch[4], branch: bestMatch[6], role: bestMatch[7], status: bestMatch[10] }};
